```sql
JOIN transactions t ON w.transaction_id = t.id
WHERE w.is_success = 0
ORDER BY w.created_at DESC
LIMIT 20;

-- –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –ø–æ –≤–µ–±—Ö—É–∫–∞–º
SELECT 
    event_type,
    COUNT(*) as total,
    SUM(is_success) as successful,
    COUNT(*) - SUM(is_success) as failed
FROM webhook_logs
GROUP BY event_type;
```
